461//-------------------------------------------- MEMORY VARIABLES ---------------------------------------------//
465// Since PSOC5 ARM memory is 4-bytes aligned, st_mem structure variables are not contiguous
466struct st_eeprom {
467
468 uint8 flag; //1 byte
469 uint8 unused_bytes[15]; //1 row (End of row 1)
470 struct st_counters cnt; //6 rows (End of row 7)
471 uint8 unused_bytes1[EEPROM_BYTES_ROW*EEPROM_AFTER_CNT_FREE_ROWS]; //3 rows (End of row 10)
472 struct st_device dev; //1 row (End of row 11)
473 struct st_motor motor[NUM_OF_MOTORS]; //7*2 rows (End of row 25)
474 struct st_encoder enc[N_ENCODER_LINE_MAX]; //3*2 rows (End of row 31)
475 struct st_emg emg; //1 row (End of row 32)
476 struct st_imu imu; //2 rows (End of row 34)
477 struct st_expansion exp; //2 rows (End of row 36)
478 struct st_user user[NUM_OF_USERS]; //2*3 rows (End of row 42)
479
480 struct st_SH_spec SH; //1 row (End of row 43)
481 struct st_JOY_spec JOY_spec; //1 row (End of row 44)
482 struct st_MASTER_spec MS; //1 row (End of row 45)
483 struct st_FB_spec FB; //1 row (End of row 46)
484 struct st_WR_spec WR; //1 row (End of row 47)
485
486 #ifdef GENERIC_FW
487 //struct st_CUFF_spec CUFF_spec;
488#endif
489};
